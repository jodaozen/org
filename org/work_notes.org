logclean ()
{
    _date=$(date +%s);
    case $1 in
        "")
            echo "Specify log name to clean.";
            return
        ;;
        *)
            l=$(echo $1)
        ;;
    esac;
    tail -n 10000 "$l" > "$l"."$_date";
    gzip "$l"."$_date";
    : > $l
}
